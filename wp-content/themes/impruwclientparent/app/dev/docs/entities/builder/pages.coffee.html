PUBLIC API FOR ENitity

 
      API =
        getPagesCollection:->
          new Pages.PageCollection

        getPages: (param = {})->
          pages = App.request "get:collection", 'pagecollection'

          if not pages
            pages = new Pages.PageCollection
            App.request "set:collection", 'pagecollection',pages
            pages.fetch
                reset : true
                data  : param

          pages

        createNewPage:(data = {})->
          page = new Pages.PageModel data
          page
