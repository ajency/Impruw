            class Media.MediaCollection extends Backbone.Collection
                
                model : Media.MediaModel

                url :(models = []) ->

                    if models.length is 0 
                        "#{AJAXURL}?action=media"
                    else
                        ids = []
                        ids.push media.get('ID') for media in models
                        ids = ids.join()
                        "#{AJAXURL}?action=media&ids=#{ids}"
